function prepareUrl(url) {
    if (url.lastIndexOf('&') == url.length - 1) {
        return url;
    }

    if (url.indexOf('?') > 0) {
        return url + '&';
    }

    return url + '?';
}

function loadRecipient(formId, identCode, delivered) {

function loadRecipientForShipment(formId, identCode, delivered) {
    $.getJSON(""+prepareUrl("recipent.do?org.apache.catalina.filters.CSRF_NONCE=905C182C0C8B41E2C1457E0CD3B54CED") + $("#" + formId).serialize(), function (jason) {
        if (jason.recipient.error == true) {
            $('#error_' + identCode).html(jason.recipient.responseMsg);
            $('#div_error_' + identCode).removeAttr("style");
            if (jason.recipient.reload == true) {
                gotoLogin();
            }
            return;
        }
        $('#tr_panrecipient').removeAttr("style");
        $('#panrecipient').html(jason.recipient.responseObject.panRecipient);
        if (delivered == 1) {
            $('#recipient_' + identCode).html(jason.recipient.responseObject.recipient);
        }
        $('#zip-link_' + identCode).remove();
        $('#zip-request-' + identCode).remove();
        reloadCaptcha();
    });
}

function fillRecipient(jason, identCode, delivered) {
    if (jason.recipient.error == true) {
        $('#error_' + identCode).html(jason.recipient.responseMsg);
        $('#div_error_' + identCode).removeAttr("style");
        if (jason.recipient.reload == true) {
            gotoLogin();
        }
        return;
    }

    if ($('#panrecipient_' + identCode).length > 0) {
        $('#tr_panrecipient_' + identCode).removeAttr("style");
        $('#panrecipient_' + identCode).html(jason.recipient.responseObject.panRecipient);
    }

    if (delivered == 1) {
        if ($('#recipient_' + identCode).length > 0) {
            $('#recipient_' + identCode).html(jason.recipient.responseObject.recipient);
            $('#tr_recipient_' + identCode).removeAttr("style");
        }
    }

    if ($('#zip-link_' + identCode).length > 0) {
        $('#zip-link_' + identCode).remove();
    }

    if ($('#zip-request-' + identCode).length > 0) {
        $('#zip-request-' + identCode).remove();
    }
}

            param = param + delim + "submitPieceId=" + spA[i].value;

                delim = "&";
            }
        }
    }
    return param;
}

function submitEDICC(pieceId) {
    var spA = [];
    var param = "";


    if (pieceId == null) {
        spA = document.getElementsByName("alertPiece");
        param = $("#parcelAlert").serialize();
    } else {
        spA = [
            {'id': pieceId},
            {'id': pieceId}
        ];
        param = $("#parcel-alert-form-" + pieceId).serialize();
    }



    param = param.replace("currentState=&", "currentState=on&")
    param = param.replace("estimatedDeliveryNotificationChosen=&", "estimatedDeliveryNotificationChosen=on&")
    param = param.replace("deliveredNotificationChosen=&", "deliveredNotificationChosen=on&")

    var delim = "&";
    for (i = 0; i < spA.length; i++) {
        param = param + delim + "submitPieceId=" + spA[i].id;
    }
    param = param + "&confirm=true"  + "&intern=" + isInternCall() + "&lang=de";

    $.ajax({
        type: "GET",
        url: ""+prepareUrl("parcel_alert.do?org.apache.catalina.filters.CSRF_NONCE=905C182C0C8B41E2C1457E0CD3B54CED"),
        async: false,
        data: param,
        success: function (jason) {

            var errors = false;
            var errorContainer = "";
            if (pieceId == null) {
                errorContainer = "#errors";
            } else {
                errorContainer = "#errors-parcel-alert-" + pieceId;
            }

            $(errorContainer).html("");


            if (pieceId == null) {
                $("#parcelAlert" + " input").removeClass("mm_alert");
            } else {
                $("#parcel-alert-form-" + pieceId + " input").removeClass("mm_alert");
            }


            if (typeof jason.errorResponse !== "undefined") {
                if (typeof jason.errorResponse.responseObject !== "undefined") {

                    if (typeof jason.errorResponse.responseObject.errors.id !== "undefined") {
                        errors = true;
                        id = jason.errorResponse.responseObject.errors.id;
                        message = jason.errorResponse.responseObject.errors.message;

                        $(errorContainer).append("<p class='mm_alert'>" + message + "</p>");


                        if (pieceId == null) {
                            $("#" + id).addClass("mm_alert");
                        } else {
                            $("#" + id + "-" + pieceId).addClass("mm_alert");
                        }
                    } else {
                        $.each(jason.errorResponse.responseObject.errors, function (index, error) {
                            errors = true;
                            id = error.id;
                            message = error.message;

                            $(errorContainer).append("<p class='mm_alert'>" + message + "</p>");


                            if (pieceId == null) {
                                $("#" + id).addClass("mm_alert");
                            } else {
                                $("#" + id + "-" + pieceId).addClass("mm_alert");
                            }
                        });
                    }
                }
            }




            if (errors == false) {
                $('#collapse-parcel-alert-' + pieceId).collapse('hide');
                $('#parcel-alert-success-' + pieceId).show();

                clearParcelAlertFields(pieceId);


            }


            return;
        }
    });

}

function updateOffers(pieceId) {

    if (pieceId == null) {
    }

    param = "&pieceId=" + pieceId;

    $.ajax({
        type: "GET",
        url: ""+prepareUrl("update_offers.do?org.apache.catalina.filters.CSRF_NONCE=905C182C0C8B41E2C1457E0CD3B54CED"),
        async: true,
        data: param,
        success: function (json) {

            var $servicesElements = $('tr[data-piece-id=' + pieceId + ']');
            var $servicesTooltipElements = $('a[data-piece-id=' + pieceId + ']');

            if (json.updateOffersInfo.responseObject.offersAvailable) {

                $servicesElements.show();

                $servicesTooltipElements
                        .attr('title', json.updateOffersInfo.responseObject.tooltip)
                        .attr('data-original-title', json.updateOffersInfo.responseObject.tooltip);

            } else {
                $servicesElements.hide();
            }


        }
    });

}

    $('#parcel-alert-form-' + pieceId + " input").val('');
    $('#parcel-alert-form-' + pieceId + " input").prop('checked', false);

}
